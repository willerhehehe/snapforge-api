Quick Start

Add the X-API-Key header to every request. That's it.

Python
Node.js
# AI agents call tools directly — no HTTP, no API keys needed
# Example: agent asks Claude Code to take a screenshot

User: Take a screenshot of https://example.com

Agent: [calls take_screenshot tool]
  url: "https://example.com"
  width: 1280
  height: 720
  output_path: "screenshot.png"

Result: Screenshot saved to screenshot.png (16KB, 1280x720)

# Works the same for QR codes, PDFs, image processing...
User: Generate a QR code for my website with rounded style

Agent: [calls generate_qr_code tool]
  data: "https://mysite.com"
  style: "rounded"
  fg_color: "#3b82f6"

Result: QR code saved to qr.png (24KB, 400x400)
# Take a screenshot
curl "https://api.snapforge.dev/api/v1/screenshot?url=https://example.com" \
  -H "X-API-Key: YOUR_KEY" \
  -o screenshot.png

# Generate a QR code
curl "https://api.snapforge.dev/api/v1/qr?data=https://mysite.com&style=rounded" \
  -H "X-API-Key: YOUR_KEY" \
  -o qr.png

# Convert HTML to PDF
curl -X POST "https://api.snapforge.dev/api/v1/pdf/from-html" \
  -H "X-API-Key: YOUR_KEY" \
  -H "Content-Type: text/html" \
  -d "<h1>Invoice #1234</h1><p>Total: $99.00</p>" \
  -o invoice.pdf
import httpx

API = "https://api.snapforge.dev/api/v1"
KEY = "YOUR_KEY"
headers = {"X-API-Key": KEY}

# Screenshot
r = httpx.get(f"{API}/screenshot",
    params={"url": "https://example.com"},
    headers=headers)
open("screenshot.png", "wb").write(r.content)

# Resize image
with open("photo.jpg", "rb") as f:
    r = httpx.post(f"{API}/image/resize",
        params={"width": 800, "output_format": "webp"},
        files={"file": f},
        headers=headers)
open("resized.webp", "wb").write(r.content)
const API = "https://api.snapforge.dev/api/v1";
const KEY = "YOUR_KEY";
const headers = { "X-API-Key": KEY };

// Screenshot
const res = await fetch(
  `${API}/screenshot?url=https://example.com`,
  { headers }

const buf = Buffer.from(await res.arrayBuffer());
await Bun.write("screenshot.png", buf);

// Generate OG image

  `${API}/og-image?title=My+Post&subtitle=Read+more`,

await Bun.write("og.png", Buffer.from(await og.arrayBuffer()));
